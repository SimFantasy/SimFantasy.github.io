Node 12 数据验证

Simz
2023-10-08
Node

数据验证有两个库 validator 和 validate.js

validator是一个字符串的验证库，使用上比较简单，但模板代码会比较多，但验证内容上较丰富。

validator.js是一个对象的验证库，使用和代码比较简单，但验证内容上不如validate 丰富，一些内容的验证需要自己写正则，但功能比较强大。

Validator

validator库仅验证和清理字符串。如果不确定输入是否为字符串，请使用 input + '' 强制转换。传递字符串以外的任何内容都会导致错误。

安装

1pnpm add validator

使用

示例使用validator来校验一个登录信息，来对username，passwod是否满足规则进行校验。

封装一个验证函数，返回isValid和error

在导出的validator中有很多验证方法，可以在官网中查看对应的说明。

 1// handle-validator.js
 2const validator = require('validator')
 3
 4const isEmpty = value => {
 5	return (
 6		!value ||
 7		(typeof value === 'object' && Object.keys(value).length === 0) ||
 8		(typeof value === 'sttring' && value.trim().length === 0)
 9	)
10}
11
12const handleValidator = data => {
13	const errors = {}
14
15	const { account, password } = data
16
17	// 验证账号是否为空
18	if (validator.isEmpty(account)) {
19		errors.account = '账号不能为空'
20	}
21
22	// 验证密码是否为空
23	if (validator.isEmpty(password)) {
24		errors.password = '密码不能为空'
25	}
26
27	// 验证账号长度
28	if (!validator.isLength(account, { min: 3, max: 20 })) {
29		errors.account = '账号长度不能小于6位且不能大于20位'
30	}
31
32	// 验证密码长度
33	if (!validator.isLength(password, { min: 6, max: 20 })) {
34		errors.password = '密码长度不能小于6位且不能大于20位'
35	}
36
37	return {
38		errors,
39		isValid: isEmpty(errors)
40	}
41}
42
43module.exports = handleValidator

调用handle-validator.js当isValid返回false时表示验证成功，继续执行后面的代码，如果isValid返回true时表示有校验失败，执行if中的代码。

 1// verifyAdmin.js
 2const adminService = require('../service/admin.service')
 3const handleValidator = require('../utils/handle-validator')
 4
 5const verifyAdmin = async (ctx, next) => {
 6	// 获取admin
 7	const admin = ctx.request.body
 8
 9	// 验证数据是否为空，是否符合校验规则
10	const { errors, isValid } = handleValidator(admin)
11
12	// 验证失败
13	if (!isValid) {
14		ctx.body = {
15			code: 400,
16			message: '验证失败',
17			errors
18		}
19		return
20	}
21
22	// 验证账号是否存在
23	const result = await adminService.isExist(admin.account)
24
25	// 验证失败
26	if (result) {
27		ctx.body = {
28			code: 400,
29			message: '账号已存在'
30		}
31		return
32	}
33
34	// 验证成功
35	await next()
36}

Validate

validate使用上需要自己来配置更多的内容来处理对应的验证规则。

安装

1pnpm add validate.js

使用

使用validate来验证学生的信息username、birthday、gender、mobile、classId 几项指标。

写一个验证方法来处理，validate是个对象验证的库，所以我们直接将对象传入即可。

 1// handle-student-validate.js
 2const validate = require('validate.js')
 3const dayjs = require('dayjs')
 4var utc = require('dayjs/plugin/utc')
 5const { Class } = require('../model/models')
 6dayjs.extend(utc)
 7
 8const handleStudentValidator = async data => {
 9	// 自定义验证规则，验证班级是否存在
10	validate.validators.classExists = async id => {
11		const result = await Class.findByPk(id)
12		// 如果不存在，返回错误信息，否则返回undefined
13		if (!result) {
14			return '不存在'
15		} else {
16			return
17		}
18	}
19	// 验证规则
20	const rules = {
21		name: {
22			presence: {
23				allowEmpty: false,
24				message: '不能为空'
25			},
26			type: 'string',
27			length: {
28				minimum: 2,
29				maximum: 10,
30				message: '长度必须在2-10之间'
31			}
32		},
33		birthday: {
34			presence: {
35				allowEmpty: false,
36				message: '不能为空'
37			},
38			datetime: {
39				dateOnly: true,
40				earliest: +dayjs.utc().subtract(50, 'y'),
41				latest: +dayjs.utc().subtract(10, 'y'),
42				message: '必须是一个有效的日期'
43			}
44		},
45		gender: {
46			presence: {
47				allowEmpty: false,
48				message: '不能为空'
49			},
50			type: 'boolean',
51			inclusion: {
52				within: [true, false],
53				message: '必须是一个布尔值'
54			}
55		},
56		mobile: {
57			presence: {
58				allowEmpty: false,
59				message: '不能为空'
60			},
61			numericality: {
62				onlyInteger: true,
63				strict: false,
64				message: '必须是数字'
65			},
66			format: {
67				pattern: /^1[3456789]\d{9}$/,
68				message: '必须是一个有效的手机号码'
69			}
70		},
71		classId: {
72			presence: {
73				allowEmpty: false,
74				message: '不能为空'
75			},
76			numericality: {
77				onlyInteger: true,
78				strict: false,
79				message: '必须是数字'
80			},
81			classExists: true
82		}
83	}
84
85	// const { name, birthday, mobile, gender, classId } = data
86	// 同步验证
87	// const result = await validate.validate(data, rules)
88	// 异步验证
89	try {
90		await validate.async(data, rules)
91	} catch (error) {
92		return error
93	}
94}
95
96module.exports = handleStudentValidator
参数说明：

presence：状态验证程序验证值是否已定义。通常只是将null和underfined作为是否定义的标准，所以还加上了allowEmpty: false，表示空字符串、空对象、空数组、仅空格字符串都不允许。

type：类型验证程序确保输入的类型正确。类型包括：array（数组）、interger（整数）、number（数值）、string（字符串）、date（日期）、boolean（布尔）。这个type中的整数、数值验证比较严格，如果为字符串的数字就无法通过验证，日期也是，必须为timestamp，所以这两项会使用别的验证方式来替代。

length：长度验证器将检查字符串的长度。可以使用以下长度约束

  • is：该值必须具有正好此长度。相当于“等于”。
  • minimum：最小值。
  • maximum：最大值。

datetime：此日期时间验证器可用于验证日期和时间。由于javascript对日期的解析非常差，所以需要有一些额外的方法来解析需要使用的值。

  • dateOnly：仅日期，如果为true，值只允许日期不能是日期时间。
  • earliest：日期不能早于此时间。
  • latest：日期不能再此时间之后。

earliest和latest都需要使用解析函数来处理对应的值。解析函数如下，在设置完解析函数后，只需要在入口页面加上改函数的引用require('./utils/validate.init')即可在验证datetime时自动触发。

 1// utils/validate.init.js
 2const validate = require('validate.js')
 3const dayjs = require('dayjs')
 4var utc = require('dayjs/plugin/utc')
 5dayjs.extend(utc)
 6
 7validate.extend(validate.validators.datetime, {
 8	/**
 9	 * 该函数会自动用于validate的datetime类型的验证
10	 * 在验证datetime时自动触发，用于将字符串转换为时间戳
11	 * 如果无法返回时间戳，则返回NaN
12	 * @param {*} value 要验证的值
13	 * @param {*} options 验证规则
14	 * @returns {number} 返回时间戳
15	 */
16	parse(value, options) {
17		let formats = ['YYYY-MM-DD HH:mm:ss', 'YYYY-M-D H:m:s', 'x']
18		if (options.dateOnly) {
19			formats = ['YYYY-MM-DD', 'YYYY-M-D', 'x']
20		}
21		return +dayjs.utc(value, formats, true)
22	},
23	/**
24	 * 用于显示错误信息时，将时间戳转换为字符串
25	 * @param {*} value 要验证的值
26	 * @param {*} options 验证规则
27	 */
28	format(value, options) {
29		let formats = 'YYYY-MM-DD'
30		if (!options.dateOnly) {
31			formats += ' HH:mm:ss'
32		}
33		return dayjs.utc(value).format(formats)
34	}
35})

inclusion： 包含，检查给定的值是否存在于由 within 选项给出的列表中。您可以在 within 中指定为列表或对象（在这种情况下，使用对象的键）。

numericality：数值型，数字验证器只允许数字。每个默认字符串使用 + 运算符强制为数字。如果不希望这样做，则可以将 noString 选项 true 设置为 禁用此行为。

关闭严格模式后，可以使用字符串的数字。

  • onlyInteger ：仅整数
  • strict：开启严格模式，不允许使用前导零，并且数字不能格式不正确。
  • greaterThan：大于，输入值必须大于此值。
  • greaterThanOrEqualTo：大于等于，输入必须大于或等于此值。
  • equalTo：等于，输入值必须完全等于此值。
  • lessThanOrEqualTo：小于等于，输入值必须小于或等于此值。
  • lessThan：小于，输入值必须小于此值。
  • divisibleBy：可整除，输入值必须可以整除此值。
  • odd：奇数，输入值必须是奇数。
  • even：偶数，数值必须是偶数。

format：格式化，格式验证器将根据您选择的正则表达式验证值。如果值不匹配，则默认消息无效。

  • pattern：正则表达式。

classExists：这是个自定义的验证函数。验证班级是否存在。自定义函数会将传过来的classId传入到此函数中进行验证，根据返回的内容进行验证。

  • 如果不存在此id则返回错误消息。
  • 否则返回underfined。
 1// 自定义验证规则，验证班级是否存在
 2validate.validators.classExists = async id => {
 3  const result = await Class.findByPk(id)
 4  // 如果不存在，返回错误信息，否则返回undefined
 5  if (!result) {
 6    return '不存在'
 7  } else {
 8    return
 9  }
10}

函数使用

因为不同的表单验证内容不一样，还是做成函数来进行验证比较好。

 1const studentService = require('../service/student.service')
 2const handleStudentValidator = require('../utils/handle-student-validator')
 3
 4// 新增学生
 5const createStudent = async (ctx, next) => {
 6  const student = ctx.request.body
 7  // 对学生信息进行验证
 8  const validate = await handleStudentValidator(student)
 9	// 如果验证失败，返回错误信息
10  if (validate) {
11    ctx.body = {
12      code: 400,
13      message: '验证失败',
14      errors: validate
15    }
16    return
17  }
18
19  const result = await studentService.create(student)
20
21  ctx.body = {
22    code: 200,
23    message: '新增成功',
24    data: result
25  }
26}

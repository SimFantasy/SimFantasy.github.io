Vue2.x ElementUI项目总结

Simz
2022-04-12
vue elementui

近一个月的Vue2 ElementUI 电商后台项目的总结。

  1. Vue2 的写法太累，data、mehtods都卸写在一起，除了用注释标记区分，结构上很是繁琐。vue3这块的Compositions API解决了这块的问题，期待后面要开始Vue3的项目。
  2. Vue东西还是很简单，主要还是需要项目来练习。
  3. 黑马的学习视频还是不行，每次都直接把代码贴出来，念一遍，很多关联概念不解释，只说是固定写法。尚硅谷的张天禹的好太多，就是需要去跟着他的内容去理解，掰得太碎。最近在看coderwhy的视频，还可以，该讲的也讲的比较细，就是上课中间中断太多。

以下为本次项目基础总结，入口、框架、增删改查都涉及到了。

Script

 1<script>
 2export default {
 3  name: 'VUser',
 4  data() {
 5    // 自定义邮箱验证规则
 6  	const checkEmail = (rule, value, callback) => {
 7      const regEmail = /^\w+@\w+(\.\w+)+$/
 8      if (regEmail.test(value)) return callback()
 9      callback(new Error('请输入正确的邮箱'))
10    }
11    // 自定义邮箱验证规则
12    const checkMobile = (rule, value, callback) => { 
13      const regMobile = /^1[34578]\d{9}$/
14      if (regMobile.test(value)) return callback()
15      callback(new Error('请输入正确的手机号码'))
16    }
17    return {
18      //添加用户对话框控制
19      addUserVisible: false,
20      // 添加用户数据
21      addUser: { 
22        username: '',
23        password: '',
24        email: '',
25        mobile: ''
26      },
27      // 添加用户验证规则
28      addUserRules: {
29        username: [
30          { require: true, message: '请输入用户名', trigger: 'blur'},
31          { min: 3, max: 16, message: '长度为3 ~ 16 个字符', trigger: 'blur'}
32        ],
33        password: [
34          { require: true, message: '请输入密码', trigger: 'blur'},
35          { min: 6, max: 24, message: '长度为6 ~ 24 个字符', trigger: 'blur'}
36        ],
37        email: [
38          { require: true, message: '请输入邮箱', trigger: 'blur'},
39          { validator: checkEmail, trigger: 'blur'}
40        ],
41        mobile: [
42          { require: true, message: '请输入手机', trigger: 'blur'},
43          { validator: checkMobile, trigger: 'blur'}
44        ]
45      }
46    }
47  },
48  methods: {
49    // 监听添加用户弹层关闭事件，重置表单项
50    addDialogClose(){
51      this.$refs.addUserRef.resetFields()
52    },
53    addUserForm(){
54      // 添加表单数据之前使用$refs的validate对表单数据进行预校验
55      this.$refs.addUserRef.validate(async valid => {
56        // 如果校验失败，跳出
57        if (!valid) return false
58        const { data: res } = await this.$http.post('users', this.addUser)
59        // 如果服务端返回的状态码不为201则弹出添加失败提示
60        if (res.meta.status !== 201){
61          return this.$message.error('用户添加失败')
62        }
63        this.$message.success('用户添加成功')
64        // 关闭弹层
65        this.addUserVisible = false
66        // 刷新用户列表
67        this.getUserList()

删除 Delete

Script

 1export default {
 2	methods: {
 3    // 删除数据事件，需将数据id传过来
 4    async delUserById(uid){
 5      // 点击删除按钮后，弹出二次确认弹层
 6      const confirmResult = await this.$confirm('此操作将永久删除该用户, 是否继续?', '提示', {
 7        confirmButtonText: '删除',
 8        cancelButtonText: '取消',
 9        type: 'danger'
10      }).catch(err => err)
11      // 点击删除按钮，进行确认判断，如果不为删除，则取消删除事件
12      if (confirmResult !== 'confirm') {
13        return this.$message.info('已取消删除')
14      }
15      // 如判断为删除，则进行删除请求将id传给服务端
16      const { data: res } = await this.$http.delete(`users/${uid}`)
17      if (res.meta.status !== 200){
18        return this.$message.error('删除用户失败')
19      }
20      this.$message.success('用户删除成功')
21      this.getUserList()
22    }
23  }
24}

Formik学习笔记之二

Simz
2022-10-18
react formik

这部分主要是示例练习。将input、textarea、select、radio、checkbox、data封装成Formik的组件，在使用时简化操作，并在几个简单的案例中使用。

‍

基础结构

FormikControl是封装的Formik组件。

FormikContainer是演示的文档。

由于主要是使用演示的文档，所以没有将组件样式一起做封装。

组件部分

FormikControl下的index.jsx主要用于根据传入的props来调用不同的组件，并将props传下去。

 1// /src/components/FormikControl/index.jsx
 2
 3import React from 'react'
 4import Checkbox from './Checkbox'
 5import DatePicker from './DatePicker'
 6import Input from './Input'
 7import Radio from './Radio'
 8import Select from './Select'
 9import Textarea from './Textarea'
10
11const FomikControl = ({ control, ...rest }) => {
12  switch (control) {
13    case 'input':
14      return <Input {...rest} />
15    case 'textarea':
16      return <Textarea {...rest} />
17    case 'select':
18      return <Select {...rest} />
19    case 'radio':
20      return <Radio {...rest} />
21    case 'checkbox':
22      return <Checkbox {...rest} />
23    case 'date':
24      return <DatePicker {...rest} />
25    default:
26      return null
27  }
28}
29
30export default FomikControl

Checkbox

checkbox的坑：

在input上的checked上需要根据fielde.value判断是否包含option.value的值，需要用Boolean包裹得到一个布尔值，给到checked

 1// src/components/FormikControl/Checkbox.jsx
 2
 3import React from 'react'
 4import { Field, ErrorMessage } from 'formik'
 5import { ErrorText } from '@/pages/FormikContainer/style'
 6
 7const Checkbox = ({ label, name, options, ...rest }) => {
 8  return (
 9    <div className='form-control'>
10      <label htmlFor={name}>{label}</label>
11      <Field name={name} {...rest}>
12        {({ field }) => {
13          return (
14            <div className='form-option-group'>
15              {options.map(option => (
16                <div className='form-option-item' key={option.key}>
17                  <input
18                    type='checkbox'
19                    id={option.value}
20                    {...field}
21                    value={option.value}
22                    checked={Boolean(field.value.includes(option.value))}
23                  />
24                  <label htmlFor={option.value}>{option.key}</label>
25                </div>
26              ))}
27            </div>
28          )
29        }}
30      </Field>
31      <ErrorMessage name={name} component={ErrorText} />
32    </div>
33  )
34}
35
36export default Checkbox

DatePicker

datepicker是用了react-datepicker这个插件，但似乎比较久远了，样式在演示中有错位，但不影响结果。

 1// src/components/FormikControl/DatePicker.jsx
 2
 3import React from 'react'
 4import { Field, ErrorMessage } from 'formik'
 5import { ErrorText } from '@/pages/FormikContainer/style'
 6import DateView from 'react-datepicker'
 7import 'react-datepicker/dist/react-datepicker.css'
 8
 9const DatePicker = ({ label, name, ...rest }) => {
10  return (
11    <div className='form-control'>
12      <label htmlFor={name}>{label}</label>
13      <Field name={name} {...rest}>
14        {({ field, form }) => {
15          const { setFieldValue } = form
16          const { value } = field
17          return (
18            <DateView
19              id={name}
20              {...field}
21              {...rest}
22              selected={value}
23              onChange={date => setFieldValue(name, date)}
24            />
25          )
26        }}
27      </Field>
28      <ErrorMessage name={name} component={ErrorText} />
29    </div>
30  )
31}
32
33export default DatePicker

演示一

需要注意的两个问题：

  1. checkbox的初始值需要为空数组。因为可以放入多个值。
  2. checkbox的验证必选，不能使用required，只能用array的min来设置最少选择一个。

 2

 5import * as Yup from 'yup'

 8

11    { key: '请选择', value: '' },
12    { key: '选项一', value: 'option1' },
13    { key: '选项二', value: 'option2' },
14    { key: '选项三', value: 'option3' }
15  ]
16  const radioOptions = [
17    { key: '选项一', value: 'radio1' },
18    { key: '选项二', value: 'radio2' },
19    { key: '选项三', value: 'radio3' }
20  ]
21
22  const checkboxOptions = [
23    { key: '选项一', value: 'checkbox1' },
24    { key: '选项二', value: 'checkbox2' },
25    { key: '选项三', value: 'checkbox3' }
26  ]
27
28  const initialValues = {
29    email: '',
30    description: '',
31    selectOption: '',
32    radioOption: '',
33    checkboxOption: [],
34    birthDate: ''
35  }

 7import { FormikContainerWrap } from '../FormikContainer/style'
 8
 9const Register = () => {
10  const options = [
11    { key: 'email', value: 'emailMoc' },
12    { key: 'phone', value: 'phoneMoc' }
13  ]
14
15  const initialValues = {
16    email: '',
17    password: '',
18    comfilmPassword: '',
19    modOfContact: '',
20    phone: ''
21  }
22
23  const validationSchema = Yup.object({
24    email: Yup.string().email('请输入正确邮箱').required('请输入邮箱'),

27      .max(12, '请输入6-12位密码')
28      .required('请输入密码'),

40    console.log('formik values: ', values)
41  }
42
43  return (

45      <Formik initialValues={initialValues} validationSchema={validationSchema} onSubmit={onSubmit}>
46        {formik => (
47          <Form>
48            <FormikControl control='input' type='email' label='邮箱' name='email' />
49            <FormikControl control='input' type='password' label='密码' name='password' />

51              control='input'
52              type='password'
53              label='确认密码'
54              name='confirmPassword'
55            />
56            <FormikControl control='radio' label='联系方式' name='modOfContact' options={options} />
57            <FormikControl control='input' type='text' label='电话' name='phone' />
58            <div className='form-control'>
59              <button
60                className='form-btn'
61                type='submit'
62                disabled={!(formik.dirty && formik.isValid)}
63              >
64                注册
65              </button>
66            </div>
67          </Form>
68        )}
69      </Formik>
70    </FormikContainerWrap>

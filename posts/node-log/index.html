Node 13 日志记录

Simz
2023-10-08
node

使用log4js来进行项目的日志记录，express和koa的morgan日志记录过于简单，很多东西无法更好的来记录。

安装：

1pnpm add log4js

几个概念

日志级别level：记录日志的等级，例如：调试日志、信息日志、错误日志等。

日志分类category：记录的内容分类，记录SQL操作的日志、请求日志等。

日志出口appender：日志记录到什么地方。是在控制台打印还是存文本。

日志书写格式layouts：日志输出内容的格式。

基本使用

1var log4js = require("log4js");
2var logger = log4js.getLogger();
3
4logger.level = "debug"; 
5logger.debug("Some debug messages");

封装日志记录函数

 1// logger.js
 2const log4js = require('log4js')
 3const path = require('path')
 4
 5// 配置log4js
 6log4js.configure({
 7	// 指定日志输出的位置及文件名、日志级别
 8	appenders: {
 9		// 定义sql日志出口
10		sql: {
11			// 日志类型,输出到文件'file'文件夹下，以日期命名文件，文件名为：sql-2021-07-29.log
12			type: 'datefile',
13			// 输出的文件名
14			filename: path.resolve(__dirname, 'logs', 'sql', 'logging.log'),
15			// 日志文件大小限制，超出后会自动新建文件，单位：字节
16			maxLogSize: 1024 * 1024 * 10,
17			// 保留文件后缀
18			keepFileExt: true,
19			// 日志文件保存天数
20			daysToKeep: 30,
21			// 输入日志模板
22			layout: {
23				// 日志输出格式类型为pattern
24				type: 'pattern',
25				// 日志输出格式, %d表示日期，%p表示日志级别，%c表示日志分类，%m表示日志内容，%n表示换行
26				pattern: '%c [%d{yyyy-MM-dd hh:mm:ss}] [%p] %m%n'
27			}
28		},
29		// 定义默认日志出口
30		default: {
31			// 日志类型,输出到控制台
32			type: 'stdout',
33			// 输出的文件名
34			filename: path.resolve(__dirname, 'logs', 'default', 'logging.log')
35		}
36	},
37	// 指定日志的默认配置项
38	categories: {
39		sql: {
40			// 指定日志使用sql的配置写入日志
41			appenders: ['sql'],
42			// 指定日志级别
43			level: 'all'
44		},
45		default: {
46			// 指定日志使用default的配置写入日志
47			appenders: ['default'],
48			// 指定日志级别
49			level: 'all'
50		}
51	}
52})
53
54// 监听进程退出，关闭log4js
55process.on('exit', () => {
56	// 关闭log4js
57	log4js.shutdown()
58})
59
60const sqlLogger = log4js.getLogger('sql')
61const logger = log4js.getLogger()
62
63module.exports = {
64	sqlLogger,
65	logger
66}

结合Sequelize配置输出对应的sql操作日志。

 1// src/db/config.js
 2const { Sequelize } = require('sequelize')
 3const { sqlLogger } = require('./logger')
 4
 5// 数据库配置
 6const db = {
 7	database: 'db_sequelize_school',
 8	user: 'root',
 9	password: '123456',
10	options: {
11		host: 'localhost',
12		dialect: 'mysql',
13		port: 3306,
14		// logging: false // 关闭打印SQL日志
15    // logging 可以执行一个函数，将日志信息返回给函数
16		logging: log => {
17			sqlLogger.debug(log)
18		}
19	}
20}
21
22// 连接数据库
23const sequelize = new Sequelize(db.database, db.user, db.password, db.options)
24
25// 测试连接
26!(async () => {
27	try {
28		await sequelize.authenticate()
29		console.log('Connection has been established successfully.')
30	} catch (error) {
31		console.error('Unable to connect to the database:', error)
32	}
33})()
34
35module.exports = sequelize

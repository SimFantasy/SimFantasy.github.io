 1// pages/api/coffees/index.js
 2
 3import coffees from '@/data/coffees'
 4
 5export default function handler(req, res) {
 6  if (req.method === 'GET') {
 7    res.status(200).json(coffees)
 8  } else if (req.method === 'POST') {
 9    const coffee = req.body.coffee
10    const newCoffee = {
11      id: Date.now(),
12      text: coffee
13    }
14    coffees.push(newCoffee)
15    res.status(201).json(newCoffee)
16  }
17}

对单独的数据进行get查询和delete删除数据

 1// pages/api/coffees/[coffeeId].js
 2
 3import coffees from '@/data/coffees'
 4
 5export default function handler(req, res) {
 6  const { coffeeId } = req.query
 7  if (req.method === 'GET') {
 8    const coffee = coffees.find(coffee => coffee.id === parseInt(coffeeId))
 9    req.status(200).json(coffee)
10  } else if (req.method === 'DELETE') {
11    const deleteCoffee = coffees.find(coffee => coffee.id === parseInt(coffeeId))
12    const index = coffees.findIndex(coffee => coffee.id === parseInt(coffeeId))
13    coffees.splice(index, 1)
14    res.status(200).json(deleteCoffee)
15  }
16}

API路由总结

  1. API路由机制类似于页面的路由机制。
  2. API路由根据文件名于路由进行关联
  3. 每个API路由都会导出一个默认函数，通常称为处理函数（handler）
  4. 处理函数接收request简称req和response简称req作为请求参数和响应参数
  5. 处理函数通过req.method接收不同的请求类型。例如：GET和POST
  6. 通过[...params].js文件命名，使用req.query.params来捕获所有路由参数

NextJs中组件的样式

‍

NextJs中可以直接使用module.css

1// styles/about.css
2
3.aboutBlock {
4  background-color: skyblue;
5  color: #444;
6}

然后在组件中进行引用。

1// pages/about.jsx
2
3import style from '@/styles/about.module.css'
4
5const About = () => {
6  return <div className={style.aboutBlock}>About</div>
7}
8
9export default About

在css.module中的优点：

  1. 可以让组件各种引用各自的样式，即使样式名称相同，在同一个组件引入不同子组件中有相同的样式，也不会发生冲突。
  2. 最容易使用，NextJs可以直接使用，无需安装或引用什么包。

‍

在css.module中的缺点：

  1. 每次都引用后，样式都要写成对象形式，比如：<div className={style.aboutBlock}>About</div>，否则不生效。导致会重复多次style这个单词。
  2. 在写法上module也只能按照对象的写法，把样式写成驼峰，不能使用中划线连接。这个就比较别扭。

‍

NextJs中使用css-in-js​

引用主题中的公共变量

在pages/_app.js中添加themeProvider，并包裹下Component，定义主题变量。

 1// pages/_app.js
 2
 3import { ThemeProvider } from 'styled-components'
 4
 5function MyApp({ Component, pageProps }) {
 6  const theme = {
 7    colors: {
 8      primary: '#0099e8'
 9    }
10  }
11
12  return (
13    <ThemeProvider theme={theme}>
14      <Component {...pageProps} />
15    </ThemeProvider>
16  )
17}
18
19export default MyApp

在页面中使用

 1import styled from 'styled-components'
 2
 3const AboutBlock = styled.div`
 4  padding: 8px;
 5  background-color: ${({ theme }) => theme.colors.primary};
 6  color: #fff;
 7`
 8
 9const About = () => {
10  return (
11    <AboutBlock>
12      <span>About</span>
13    </AboutBlock>

这种方法适用于定义多主题的时候，平常用的比较少，主要在使用变量时需要用一个方法从theme中传对应的变量过来，过于麻烦。平时使用的较多的是从文件中导入变量。

NextJs中的布局嵌套

在NextJs中，可以通过在pages/_app.js这个公共布局组件上添加布局响应的头尾组件。但有一些页面可能不需要公共的组件，则需要通过getLayout这个方法进行处理。

 1// pages/_app.js
 2
 3import '../styles/globals.css'
 4import Header from '@/components/layout/Header'
 5import Footer from '@/components/layout/Footer'
 6
 7function MyApp({ Component, pageProps }) {
 8  if (Component.getLayout) {
 9    return Component.getLayout(<Component {...pageProps} />)
10  }
11  return (
12    <>
13      <Header />
14      <Component {...pageProps} />
15      <Footer />
16    </>
17  )
18}
19
20export default MyApp

上述代码会通过getLayout进行判断，如果有这个方法的页面，就使用页面定义的Layout。

 1// pages/about.jsx
 2
 3import Header from '@/components/layout/Header'
 4import React from 'react'
 5import styled from 'styled-components'
 6
 7const About = () => {
 8  return (
 9    <div>
10      <span>About</span>
11    </div>
12  )
13}
14
15export default About
16
17About.getLayout = page => {
18  return (
19    <>
20      <Header />
21      {page}
22    </>
23  )
24}

上述页面，就没有使用<Footer />这个底部组件。也可以引用其他的头尾组件。

NextJs中的Head组件

作为多页面应用，每个页面都会需要一个Head标签，定义这个页面的一些Meta内容，比如：页面标题、Ico、描述等。NextJs可以在公共模板pages/_app.js中使用这个公共配置。

 1// pages/_app.js
 2
 3import Head from 'next/head'
 4import '../styles/globals.css'
 5import Header from '@/components/layout/Header'
 6import Footer from '@/components/layout/Footer'
 7
 8function MyApp({ Component, pageProps }) {
 9  if (Component.getLayout) {
10    return Component.getLayout(<Component {...pageProps} />)
11  }
12  return (
13    <>
14      <Head>
15        <title>Sim Site</title>
16        <meta name='discription' content='这个是页面描述' />
17      </Head>
18      <Header />
19      <Component {...pageProps} />
20      <Footer />
21    </>
22  )
23}
24
25export default MyApp

NextJs中的图片

在NextJs中引入图片，可以像普通的图片一样通过import的方式引入。

NextJs也提供了一个图片的API，可以对图片进行优化处理。

对于图片有两种引用方式，一种是本地图片，一种是远程图片。

如果引用远程图片的话，还需要配置下next.config.js，在配置中加入图片的主域。

NextJs中的模块导入别名设置

在NextJs中可以新建一个jsconfig.json或tsconfig.json文件来配置路径别名。

 1// jsconfig.json
 2
 3{
 4  "compilerOptions": {
 5    "baseUrl": ".",
 6    "paths": {
 7      "@/*": ["/*"]
 8    }
 9  }
10}

如果配置了"baseUrl": "."，路径其实就可以使省去前面的相对路径层级。比如：components文件夹，在pages/posts/index.jsx中引用。可以直接写components/post。但是为了书写习惯和React中保持一致，也可以配置paths如上方一样。

NextJs中的重定向

在NextJs中可以将一些旧的内容重定向到一个新的页面或组件上。在next.config.js中配置

 1// next.config.js
 2
 3const nextConfig = {
 4  redirects: async () => {
 5    return [
 6      {
 7        source: '/about',
 8        destination: '/',
 9        permanent: true
10      },
11      {
12        source: '/posts2/:postId',
13        destination: '/posts/:postId',
14        permanent: true
15      }
16    ]
17  }
18}

source：源路径，可以使用正则等方法对路径进行匹配。

destination：跳转到的路径

permanent：是否长期重定向，主要给搜索引擎查看在headers上的重定向内容。

NextJs中的环境变量配置

‍

在根目录下添加.env.local文件。文件中可以配置数据库、外部API的内容。通过process.env在代码中取出。

1// .env.local
2
3DB_HOST=localhost
4DB_USER=myuser
5DB_PASS=mypassword

‍

1// pages/index.js
2
3export async function getStaticProps() {
4  const db = await myDB.connect({
5    host: process.env.DB_HOST,
6    username: process.env.DB_USER,
7    password: process.env.DB_PASS,
8  })
9}

默认情况下，环境变量仅可在node环境中使用，不会暴露给浏览器。

如果需要将环境变量内容暴露给浏览器可以在变量前增加前缀NEXT_PUBLIC_，这样就可以在浏览器中访问到该变量。

1// .env.local
2
3NEXT_PUBLIC_ANALYTICS_ID=abcdefghijk

‍

 1// pages/index.js
 2
 3import setupAnalyticsService from '../lib/my-analytics-service'
 4
 5setupAnalyticsService(process.env.NEXT_PUBLIC_ANALYTICS_ID)
 6
 7function HomePage() {
 8  return <h1>Hello World</h1>
 9}
10
11export default HomePage

‍

在通常情况下只有一个环境变量，但有时会对开发环境和生产环境进行区分。所以可以进行如下配置：

  1. .env.local可以覆盖所有默认值，可用来保存一些公共的信息。另外该文件不会被提交到源码仓库，可以放置一些密钥类的信息。
  2. .env.development设置开发环境默认值
  3. .env.production设置生产环境默认值

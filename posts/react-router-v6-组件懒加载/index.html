React Router v6 组件懒加载

Simz
2022-07-27
react react-router

还是React Router V6，在v5的时候组件懒加载只需要在路由页面使用lazy方法import组件后，然后在App组件外层包一层Suspense后就可以实现了，在v6里就搞得很麻烦，因为Suspense包在了每个路由的外面。

V5与V6使用比较

 1// v5
 2
 3// routes里
 4const routes = [
 5  {
 6    path: '/user/home',
 7    component: React.lazy(() => import('@/pages/home')) // 组件懒加载
 8  },
 9  ...
10]
11
12// Suspense包裹在APP中包裹路由就可以了
13function App() {
14    return (
15        <HashRouter>
16            <Suspense fallback={<LazyLoading />}>
17                <ContextProvider>{renderChildren(routes)}</ContextProvider>
18            </Suspense>
19        </HashRouter>
20    );
21}
22
23// v6
24
25const HomeLazy = lazy(() => import('@/pages/home'))  // 组件懒加载
26
27const routes = [
28  {
29    path: 'home',
30    element: (
31        <Suspense fallback={<LazyLoading />}>  // Suspense包裹在了组件懒加载的外面
32          <HomeLazy />
33        </Suspense>
34    )
35  }
36  ...
37]

V6中组件懒加载方法

这就搞的很麻烦，不但代码量增加，这样搞得每个element都需要包裹一个Suspense，体力活增加了。

借用掘金佟舟大神的方法，写起来就就简单了。

 1// /src/router/index.jsx
 2
 3import React, { lazy, Suspense } from 'react'
 4import { Navigate, useRoutes } from 'react-router-dom'
 5
 6const routes = [
 7  {
 8    path: '/',
 9    element: () => import('@/pages/Home'),
10    children: [
11      { path: '/', element: <Navigate to='/discover' /> },
12      {
13        path: 'discover',
14        element: () => import('@/pages/Home/Discover'),
15        children: [
16          { index: true, element: () => import('@/pages/Home/Discover/Recommend')},
17          { path: 'recommend', element: () => import('@/pages/Home/Discover/Recommend') },
18          { path: 'playlist', element: () => import('@/pages/Home/Discover/Playlist') },
19          { path: 'djradio', element: () => import('@/pages/Home/Discover/Djradio') },
20          { path: 'rank', element: () => import('@/pages/Home/Discover/Rank') },
21          { path: 'artists', element: () => import('@/pages/Home/Discover/Artists') },
22          { path: 'latesmusic', element: () => import('@/pages/Home/Discover/LatesMusic') },
23        ]
24      },
25      { path: '/video', element: () => import('@/pages/Home/Video') },
26      { path: '/friend', element: () => import('@/pages/Home/Friend') },
27    ]
28  },
29  { path: '/login', element: () => import('@/pages/Login') },
30  { path: '/search', element: () => import('@/pages/Search') },
31]
32
33function LazyElement({ importFunc }) {
34  const LazyComponent = lazy(importFunc)
35  return (
36    <Suspense fallback={<div>Loading...</div>}>
37      <LazyComponent />
38    </Suspense>
39  )
40}
41
42// element如果是懒加载，则包裹Suspense
43(function dealRoutes(routesArr) {
44  if (routesArr && Array.isArray(routesArr) && routesArr.length > 0) {
45    routesArr.forEach((route) => {
46      if (route.element && typeof route.element == 'function') {
47        const importFunc = route.element
48        route.element = <LazyElement importFunc={importFunc} />
49      }
50      if (route.children) {
51        dealRoutes(route.children)
52      }
53    })
54  }
55})(routes)
56
57
58export default function RootRoutes() {
59  const element = useRoutes(routes)
60  return element
61}

以上就完成组件懒加载，完美~
